//判断奇数还是偶数进行加法
#include<stdio.h>
int main()
{
    int t;
    scanf ("%d",&t);
    while(t--)
    {
        int n;
        int s1=0,s2=0,a;
        scanf ("%d",&n);
        while(n--)                          //promote myself
        {
            scanf ("%d",&a);                //+1
            if (a%2)                        //1 is true
                s1+=a;
            else
                s2+=a;
        }
        printf ("%d %d\n",s2,s1);
    }
    return 0;
}
