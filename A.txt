#include<stdio.h>
int main()
{
    int a,c;
    scanf("%d",&a);
    while(a--)
    {
        char b[200];
        scanf("%s %s %s %s",b,b,b,b);
        for(c=0;c<200;c++)
        {
            if(b[c]=='.')
                b[c]=0;
        }
        printf("Hello,%s!\n",b);
    }
    return 0;
}
