Description
这里有n个整数b1，b2……bn，和另外n个整数a1，a2……an给你一个等式

ai=bi-b(i+1)+b(i+2)-b(i+3)....一直到第n个数，现在给你a1,a2……an的值让你求出b1,b2……bn的值

Input
输入一个正整数t代表有t组数据t<100

然后一个正整数n<100

然后是a1，a2……an

保证所涉及的数都在int范围内

Output
每组数据按顺序输出n个数代表b1，b2，……bn

Sample Input
1
5
6 -4 8 -2 3
Sample Output
2 4 6 1 3


#include<stdio.h>
#include<algorithm>
#include<iostream>
#include<string.h>
#include<math.h>
using namespace std;
#define N 100005
int a[N],b[N];
int main()
{
    int t;
    scanf ("%d",&t);
    while(t--)
    {
        int n;
        scanf ("%d",&n);
        for (int i=1;i<=n;i++)
            scanf ("%d",&a[i]);
        for (int i=n;i>=1;i--)
        {
            if (i==n)
                b[i]=a[i];
            else//倒推一下
                b[i]=a[i]+a[i+1];
        }
        for (int i=1;i<=n;i++)
        {
            printf ("%d",b[i]);
            if (i!=n)
                printf (" ");
        }
        printf ("\n");
    }
    return 0;
}
