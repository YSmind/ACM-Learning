#include<stdio.h>
int main()
{
    int a=1,b=1,c,d,N,f;
    scanf("%d",&N);
    if(N==1)
        printf("1");
    else if(N==2)
        printf("1 1");
    else
    {
        printf("1 1");
        N-=2;
        f=0;
        while(N--)
        {
            f++;
            if(f%2)
            {
                a=a+b;
                printf(" %d",a);
            }
            else
            {
                b=a+b;
                printf(" %d",b);
            }
        }
    }
    return 0;
}
