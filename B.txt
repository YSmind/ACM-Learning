一道简单的数学入门题目，叫做斐波那契数列。

这个数列你们肯定见过，或者听过，没听过我在这里告诉你。

这个数列的是这样的，1,1,2,3,5,8,13,21
Input
输入一个数字T,表示T次询问，接下来没次输入一个数n（n<55），请输出第n个斐波那契数。

3

1

2

3

Output
1

1

2


#include<stdio.h>
int main()
{
    long long int a,b,c,d;
    long long int C[55];
    a=0;
    b=1;
    C[0]=0;
    C[1]=1;
    for(c=2;c<55;c++)
    {
        C[c]=C[c-1]+C[c-2];
    }
    scanf("%d",&d);
    while(d--)
    {
        int e=0;
        scanf("%d",&e);
        printf("%d\n",C[e]);
    }
    return 0;
}
