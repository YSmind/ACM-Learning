Input
输入20个整数，数字范围在1到20之间

Output
找出众数！（简单吧）

虽然题目简单，但是做法却有很多种，想出一个好的解法，也可以减少你的代码行数。






#include<stdio.h>
int main()
{
    int n[20],N[20],M[20],a,b,c;
    for(a=0;a<20;a++)//input n[]
    {
        scanf("%d",&n[a]);
    }
    for(a=0;a<20;a++)//make N[]
        N[a]=0;
    for(a=0;a<20;a++)//统计 in N[]
    {
        for(b=1;b<21;b++)
        {
            c=b-1;
            if(n[a]==b)
               N[c]++;
        }
    }
    for(a=0;a<20;a++)//备份 in M[]
    {
        M[a]=N[a];
    }
    int i,j,o;
    for(i=0;i<20;i++)//冒泡排序
    {
        for(j=i+1;j<20;j++)
        {
            if(N[i]>N[j])
            {
                o=N[i];
                N[i]=N[j];
                N[j]=o;
            }
        }
    }
    b=N[19];
    for(a=0;a<20;a++)//检测原数组中最大值
    {
        if(M[a]==b)
            printf("%d",a+1);
    }
    return 0;
}
