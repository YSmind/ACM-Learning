//case语句会比较简单 注意1 10 15 20 25 这种情况
#include<stdio.h>
int main()
{
    int s,a,b;
    int t;
    scanf ("%d",&t);
    while(t--)
    {
        scanf ("%d",&s);
        switch(s)
        {   case 0:printf ("zero\n");continue;
            case 10:printf ("ten\n");continue;
            case 11:printf ("eleven\n");continue;
            case 12:printf ("twelve\n");continue;
            case 13:printf ("thirteen\n");continue;
            case 14:printf ("fourteen\n");continue;
            case 15:printf ("fifteen\n");continue;
            case 16:printf ("sixteen\n");continue;
            case 17:printf ("seventeen\n");continue;
            case 18:printf ("eighteen\n");continue;
            case 19:printf ("nineteen\n");continue;
        }
        a = s /10;
        b = s%10;
        switch(a)
        {
            case 0:break;
            case 2:printf ("twenty");break;
            case 3:printf ("thirty");break;
            case 4:printf ("forty");break;
            case 5:printf ("fifty");break;
            case 6:printf ("sixty");break;
            case 7:printf ("seventy");break;
            case 8:printf ("eighty");break;
            case 9:printf ("ninety");break;
        }
        if (b!=0)
        {
            if (a!=0)
            printf ("-");
            switch(b)
        {
            case 1:printf ("one");break;
            case 2:printf ("two");break;
            case 3:printf ("three");break;
            case 4:printf ("four");break;
            case 5:printf ("five");break;
            case 6:printf ("six");break;
            case 7:printf ("seven");break;
            case 8:printf ("eight");break;
            case 9:printf ("nine");break;
        }
        }
        printf ("\n");
    }
    return 0;
}
