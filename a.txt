每组数据的输出有一行："Hello,XXX!"，代表JLL的回答，其中XXX表示Jonariguez的名字。



#include <stdio.h>
#include <string.h>

char name[1200];	//我在这开到1200，一点事没有，还可以开到100000呢

int main()
{

    scanf("%d",&T);
    while(T--){
        scanf("%s",name);
        scanf("%s",name);
        scanf("%s",name);
        scanf("%s",name);
        n=strlen(name);
