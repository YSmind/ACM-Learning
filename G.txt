#include<stdio.h>
int main()
{
    int t,n,m;
    scanf("%d",&
          t);
    while(t--)
    {
        scanf("%d %d",&n,&m);
        int a,b;
        if(m<=n/2)
            printf("%d",m);
        else if(n<=m/2)
            printf("%d",n);
        else
        {
            if(n%2==0&&m%2==0)
            {
                a=(2*n-m)/3;
                b=(2*m-n)/3;
                printf("%d\n",a+b);
            }
            else if(n%2==1&&m%2==1)
            {
                n--;
                m--;
                a=(2*n-m)/3;
                b=(2*m-n)/3;
                int N,M;
                N=n-(2*a+b);
                M=m-(2*b+a);
                if(N==1||M==1)
                    printf("%d\n",a+b+1);
                else
                    printf("%d\n",a+b);
            }
            else if(n%2==0&&m%2==1)
            {
                m--;
                a=(2*n-m)/3;
                b=(2*m-n)/3;
                int N,M;
                N=n-(2*a+b);
                M=m-(2*b+a);
                if(N==1&&M==1)
                    printf("%d\n",a+b+1);
                else
                    printf("%d\n",a+b);
            }
            else if(n%2==1&&m%2==0)
            {
                n--;
                a=(2*n-m)/3;
                b=(2*m-n)/3;
                int N,M;
                N=n-(2*a+b);
                M=m-(2*b+a);
                if(N==1&&M==1)
                    printf("%d\n",a+b+1);
                else
                    printf("%d\n",a+b);
            }
        }
    }
    return 0;
}
