#include<stdio.h>
int main()
{
    int n,m;
    scanf ("%d%d",&n,&m);
    int sum=0;
    for (int i=0;i<n;i++)
        for (int j=0;j<m;j++)   //the means to disposed matrix
    {
        int p;
        scanf ("%d",&p);
        if (p==1)
            sum++;              //the form is permitted
    }
    printf ("%d",sum);
    return 0;
}
